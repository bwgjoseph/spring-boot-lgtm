# 🚀 Observability Stack Quick Reference (PowerShell)

# 1. Add Repos & Update
helm repo add grafana https://grafana.github.io/helm-charts; helm repo add grafana-community https://grafana-community.github.io/helm-charts; helm repo add prometheus-community https://prometheus-community.github.io/helm-charts; helm repo update

# 2. Deploy Infrastructure
helm upgrade --install minio minio/minio -n monitoring -f values-minio.yaml --version 5.4.0
helm upgrade --install prometheus prometheus-community/prometheus -n monitoring -f values-prometheus.yaml --version 28.13.0
helm upgrade --install loki grafana-community/loki -n monitoring -f values-loki-scalable.yaml --version 9.3.4
helm upgrade --install tempo grafana-community/tempo -n monitoring -f values-tempo-scalable.yaml --version 2.0.0
helm upgrade --install grafana grafana-community/grafana -n monitoring -f values-grafana.yaml --version 11.3.0
kubectl create configmap local-dashboards --from-file=dashboards -n monitoring --dry-run=client -o yaml | kubectl apply -f -; kubectl label configmap local-dashboards grafana_dashboard=1 -n monitoring --overwrite
helm upgrade --install alloy grafana/alloy -n monitoring -f values-alloy.yaml --version 1.6.1

# 3. Build & Deploy App
.\mvnw clean compile jib:dockerBuild
kubectl apply -f deployment.yaml; kubectl rollout restart deployment spring-boot-app -n monitoring

# 4. Access UIs (Port-Forward)
# Run these in separate shells or in the background
kubectl port-forward svc/grafana 3000:80 -n monitoring
kubectl port-forward svc/spring-boot-app-svc 8080:8080 -n monitoring
kubectl port-forward svc/prometheus-alertmanager 9093:9093 -n monitoring
kubectl port-forward daemonset/alloy 12345:12345 -n monitoring

# 5. Credentials
kubectl get secret --namespace monitoring grafana -o jsonpath="{.data.admin-password}" | base64 --decode ; echo ""
